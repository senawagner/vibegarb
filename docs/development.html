🚀 Desenvolvimento

Guia completo de desenvolvimento para o Vibe Garb E-commerce. Este documento fornece workflows detalhados, práticas recomendadas, comandos essenciais e procedimentos para desenvolvimento eficiente. Inclui setup de ambiente, estrutura de código, debugging, testing e deploy. Essencial para desenvolvedores que trabalham no projeto, desde configuração inicial até implementação de novas funcionalidades.

Atualizado: 07/07/2025 - Guia de Desenvolvimento v1.0.0

⚡ Quick Start

Ambiente: Docker + Laravel + MySQL + Redis

Porta de Desenvolvimento: http://localhost:8500

Credenciais Admin: admin@vibegarb.com / admin123

Comandos Essenciais

Subir ambiente: docker-compose up -d
Ver logs: docker-compose logs -f app
Executar migrations: docker exec vibegarb_app php artisan migrate
Build assets: docker exec vibegarb_node npm run build
Limpar cache: docker exec vibegarb_app php artisan optimize:clear
Acessar tinker: docker exec -it vibegarb_app php artisan tinker

🛠️ Setup de Ambiente

Pré-requisitos

  • 🐳 Docker Desktop: Versão 4.0+ com Docker Compose
  • 📝 Editor: VS Code com extensões PHP/Laravel
  • 🌐 Git: Para controle de versão
  • 💻 Terminal: PowerShell (Windows) ou Terminal (macOS/Linux)

Primeira Instalação

# 1. Clonar repositório
git clone <repository-url>
cd vibegarb

# 2. Configurar ambiente
cp .env.example .env

# 3. Configurações do .env
APP_NAME="Vibe Garb"
APP_ENV=local
APP_DEBUG=true
APP_URL=http://localhost:8500

DB_CONNECTION=mysql
DB_HOST=db
DB_PORT=3306
DB_DATABASE=vibegarb
DB_USERNAME=vibegarb
DB_PASSWORD=secret

CACHE_DRIVER=redis
SESSION_DRIVER=file
QUEUE_CONNECTION=sync

REDIS_HOST=redis
REDIS_PASSWORD=null
REDIS_PORT=6379

# 4. Subir containers
docker-compose up -d

# 5. Instalar dependências
docker exec vibegarb_app composer install
docker exec vibegarb_node npm install

# 6. Configurar Laravel
docker exec vibegarb_app php artisan key:generate
docker exec vibegarb_app php artisan migrate --seed

# 7. Build assets
docker exec vibegarb_node npm run build

# 8. Otimizar configurações
docker exec vibegarb_app php artisan optimize

# 9. Verificar instalação
curl -I http://localhost:8500

📋 Model (Eloquent)

class Product extends Model
{
    protected $fillable = ['name', 'price', 'category_id'];
    
    protected $casts = [
        'price' => 'decimal:2',
        'is_active' => 'boolean',
    ];
    
    // Relacionamentos
    public function category()
    {
        return $this->belongsTo(Category::class);
    }
    
    // Business Logic
    public function getCurrentPriceAttribute()
    {
        return $this->sale_price ?? $this->price;
    }
    
    // Query Scopes
    public function scopeActive($query)
    {
        return $query->where('is_active', true);
    }
}

🎮 Controller

class ProductController extends Controller
{
    public function index(Request $request)
    {
        $products = Product::active()
            ->with(['category', 'productImages'])
            ->when($request->search, function($query, $search) {
                $query->where('name', 'like', "%{$search}%");
            })
            ->when($request->category, function($query, $category) {
                $query->where('category_id', $category);
            })
            ->paginate(12);

        $categories = Category::orderBy('name')->get();

        return view('products.index', compact('products', 'categories'));
    }
    
    public function show(Product $product)
    {
        $product->load(['category', 'productImages', 'variants']);
        
        return view('products.show', compact('product'));
    }
}

⚙️ Services e Lógica de Negócio

Padrão Service Layer

Para lógica complexa que não pertence aos Models ou Controllers, utilizamos Services:

// app/Services/CartService.php
class CartService
{
    public function addItem(Product $product, array $options = []): void
    {
        $sessionKey = 'shopping_cart';
        $cart = session()->get($sessionKey, []);
        
        $itemKey = $this->generateItemKey($product, $options);
        
        if (isset($cart[$itemKey])) {
            $cart[$itemKey]['quantity'] += $options['quantity'] ?? 1;
        } else {
            $cart[$itemKey] = [
                'product_id' => $product->id,
                'name' => $product->name,
                'price' => $product->current_price,
                'quantity' => $options['quantity'] ?? 1,
                'size' => $options['size'] ?? null,
                'color' => $options['color'] ?? null,
            ];
        }
        
        session()->put($sessionKey, $cart);
    }
    
    public function getContents(): array
    {
        return session()->get('shopping_cart', []);
    }
    
    public function calculateTotals(): array
    {
        $cart = $this->getContents();
        $subtotal = 0;
        
        foreach ($cart as $item) {
            $subtotal += $item['price'] * $item['quantity'];
        }
        
        $shipping = $subtotal >= 200 ? 0 : 15.00;
        $pixDiscount = $subtotal * 0.05; // 5% desconto PIX
        
        return [
            'subtotal' => $subtotal,
            'shipping' => $shipping,
            'pix_discount' => $pixDiscount,
            'total' => $subtotal + $shipping,
            'total_with_pix' => $subtotal + $shipping - $pixDiscount,
        ];
    }
}

// Como usar no Controller:
class CartController extends Controller
{
    public function __construct(
        private CartService $cartService
    ) {}
    
    public function store(Request $request, Product $product)
    {
        $this->cartService->addItem($product, $request->validated());
        
        return response()->json(['success' => true]);
    }
}

document.addEventListener('DOMContentLoaded', function() {
    // Busca de CEP
    const cepInput = document.getElementById('zipcode');
    if (cepInput) {
        cepInput.addEventListener('blur', function() {
            const cep = this.value.replace(/\D/g, '');
            if (cep.length === 8) {
                fetchAddressByCep(cep);
            }
        });
    }
    
    // Adicionar ao carrinho via AJAX
    const addToCartButtons = document.querySelectorAll('.add-to-cart');
    addToCartButtons.forEach(button => {
        button.addEventListener('click', function(e) {
            e.preventDefault();
            addToCart(this.dataset.productId);
        });
    });
});

async function fetchAddressByCep(cep) {
    try {
        const response = await fetch(`https://viacep.com.br/ws/${cep}/json/`);
        const data = await response.json();
        
        if (!data.erro) {
            document.getElementById('address').value = data.logradouro;
            document.getElementById('neighborhood').value = data.bairro;
            document.getElementById('city').value = data.localidade;
            document.getElementById('state').value = data.uf;
        }
    } catch (error) {
        console.error('Erro ao buscar CEP:', error);
    }
}

class ProductBrowsingTest extends TestCase
{
    public function test_user_can_browse_products()
    {
        $category = Category::factory()->create(['name' => 'Camisetas']);
        $product = Product::factory()->create([

            'category_id' => $category->id,
            'is_active' => true

        $response = $this->get('/products');
        
        $response->assertStatus(200);
        $response->assertSee('Camiseta Teste');
        $response->assertSee('Camisetas');
    }
    
    public function test_user_can_filter_products_by_category()
    {
        $category1 = Category::factory()->create(['name' => 'Camisetas']);
        $category2 = Category::factory()->create(['name' => 'Moletons']);
        
        $product1 = Product::factory()->create(['category_id' => $category1->id]);
        $product2 = Product::factory()->create(['category_id' => $category2->id]);

        $response->assertStatus(200);
        $response->assertSee($product1->name);
        $response->assertDontSee($product2->name);
    }
}

foreach (Product::with('category')->get() as $product) {
    echo $product->category->name; // Apenas 2 queries total
}

// Model Product
public function scopeActiveWithDetails($query)
{
    return $query->where('is_active', true)
                 ->with(['category', 'productImages'])
                 ->orderBy('name');
}
