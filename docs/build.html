      - name: Install PHP dependencies
        run: composer install --optimize-autoloader --no-dev
        
      - name: Install Node dependencies
        run: npm install
        
      - name: Build assets
        run: npm run build
        
      - name: Run tests
        run: php artisan test
        
      - name: Upload build artifacts
        uses: actions/upload-artifact@v4
        with:
          name: build-assets
